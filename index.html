How to get Started

Installing flutter persist http

Add this to your package's pubspec.yaml file:
dependencies:
       flutter_persist_http:
       path_provider:

getApplicationDocumentsDirectory().then((dir){
      PersistHttp(dir, "api name")
        .saveHttpGet("Your url http ")
        .then((response) {
      print(response);
     });
   });
